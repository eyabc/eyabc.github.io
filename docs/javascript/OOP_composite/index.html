다중 계층의 문제#

const viewmodel = ViewModel.get({
wrapper: ViewModel.get({styles: {width: '50%', background: '#ffa', cursor: 'pointer'}}),
title: ViewModel.get({properties: {innerHTML: 'Title'}}),
contents: ViewModel.get({properties: {innerHTML: 'Contents'}}),
});
  • 파일시스템 : 폴더안에 파일이 폴더안에 폴더
  • 알고리즘 문제에서는 동적계획법, 트리탐색

속도는 동적계획법이 더 빠르지만 정형화된 알고리즘에 가면 트리탐색이 더 쉽다.

Composite#

내 문제를 스스로 다 해결하지 않고, 위임을 반복해서 그것들을 취합하는 행위

상위 ViewModel 안의 하위 ViewModel 에 변화가 일어났을 때,

  • 상위 ViewModel 의 변화는 자기를 구독 Observer Listen 하고 있는 대상에게 notify 된다.

하위 ViewModel 변화를, 상위 ViewModel 이 Listen 구독하게 만들 고자 한다.

Binder 는 최상위 ViewModel 을 알면 모든 ViewModel 을 다 매핑할 수 있다.

  • Binder 는 최상위 ViewModel 와 계약을 하고, 하위 ViewModel 과는 계약하지 않는다.
  • 하위 ViewModel 의 변화는
    • 자신의 리스너에게도 알려야 하고,
    • 상위의 리스너에게도 알려야 한다.

subKey#

변화 한 것이 이 나 자신인지, 하위 뷰모델인지를 구분한다.

모든 자식들이 고유한 서브키를 갖고 있다고 가정했다.

const ViewModelValue = class {
subKey;
cat;
k;
v;
constructor(subKey, cat, k, v) {
this.subKey = subKey;
this.cat = cat;
this.k = k;
this.v = v;
Object.freeze(this);
}
};

ViewModel#

const ViewModel = class extends ViewModelListener {
// ...
subKey = '';
parent = null;
// 하위 ViewModel 이 변화하면 this isUpdated 에 저장한다.
viewModelUpdated(updated) {
updated.forEach(v => this.#isUpdated.add(v));
}
constructor(checker, data, _ = type(data, 'object')) {
super();
// ViewModel 일 때 k 는 [wrapper, title, contents], v 는 ViewModel 이다.
Object.entries(data).forEach(([k, v]) => {
if ('styles,attributes,properties'.includes(k)) {
// ..
} else { // << 'styles,attributes,properties' 에 해당하지 않거나 viewModel 일 때 >>
Object.defineProperty(this, k, {
enumerable: true, get: _ => v, set: newV => {
v = newV;
// 내가 보고 할 때는 나의 subKey 를 보고 하면 그만이다.
// 부모에서 수신했을 때는 자식객체 입장이기도 하다.....
// 서브키는 자식이자 자기자신이다.
// 빙빙도는 것 같지만 숙련이 필요하다.
this.#isUpdated.add(new ViewModelValue(this.subKey, '', k, v));
},
});
/*
<< 하위 ViewModel 이 존재할 때 추가적으로>>
하위 ViewModel 라면, 자신의 변화를 상위 ViewModel 에게 전달(전파)해야 한다.
상위 ViewModel 라면, 하위 ViewModel 의 변화를 수신해야 한다.
- 상위 ViewModel(서브젝트) 가 Observer 가 되어야 한다.
- 서브젝트가 리스너가 아닐 때 까지 변화가 전파되어야 한다.
- 외부에 전파할 때, subKey(자신인지 하위 ViewModel 의 변화인지를 구분) 를 알린다.
*/
// 내 자식의 변화인 것을 알려면, 내가 아닌 래퍼/타이틀/ 이 일으킨 변화라는 것을 알아야 한다.
// 래퍼의 스타일스의 백그라운드의 변화를 알아야지만 반응을 할 수 있다.
// 두번째로 원래는 뷰모델은 부모자식개념이 없지만, 이 개념이 생겼다
// 부모사정에서 서브키를 할당해 주었다. 자식이 변화했을 때 부모에게 nofify 된다.
if (v instanceof ViewModel) {
// 서브에서는 부모를 알게된다.
// 자식의 서브키가 뭔지는 부모 입장이다."래퍼/타이틀" 할당
// 자식이 누구의 소속일 때 할당된다.
// 이벤트 리스너에게 vm 을 주는데, 그 vm 은 자기 자신 vm 이 될 것이다.
// 통합생성을 못하기 때문에??그전에는 모든 이벤트 리스너에
// 자식이 뭐가오든 탑 vm 을 넣는 것을 넣었는데
// 이제는 그렇게 안되기 때문에 나밖에 모른다.
// 내 부모를 타고가서 알수 있다. 역참조하는 래퍼런스를 걸어준다.
v.parent = this;
v.subKey = k;
// 부모는 자식의 리스너가 된다.
v.addListener(this);
}
}
});
// 5) 첫번째 데이터가 완성되고나면 변화가 없지만 (아직 setter 가 한번도 안일어남)
// 새로만들어졌기 때문에 notify 를 한번 때려야한다.
// notify 가 속성이 바뀔 때 마다 때리면 바인더의 렌더가 미친듯이 일어날 것이다.
// 렌더라는 것은 어차피 requestAnimationFrame 를 초과해서 일어나지 않는다.
// 따라서 requestAnimationFrame 당 변화를 취합해서 떄리면 된다.
// 그러면 뷰모델 레벨에서 하나의 requestAnimationFrame 하나만 걸고 거기에 한거번에 조회해서 모든 notification 을 한꺼번에 때리면된다.
// notify 의 정적클래스에 한번 부탁하는 것이다. ...
ViewModel.notify(this);
Object.seal(this);
}
}

서브젝트이자 리스너인 옵저버가 대부분이다. 그러기 때문에 뷰 모델을 뷰모델리스너를 상속받도록 개조했다.

const ViewModel = class extends ViewModelListener {
static;
#subjects = new Set;
static;
#inited = false;
// 옵저버에게 통보할 서브젝트들의 모임이다.
// 내부에서는 타입체크를 할 필요 없다.
static notify(vm) {
// inited 가 필요한 이유는
// requestAnimationFrame 을 한번 발동하면 계속 발동하니깐. 최초로 발동하지말지 결정?
this.#subjects.add(vm);
if (this.#inited) return;
this.#inited = true;
// notify 가 한프레임당 한번만 호출 될 것이다.
const f = _ => {
this.#subjects.forEach(vm => {
// 업데이트 됐다는 의
if (vm.#isUpdated.size) {
vm.notify();
vm.#isUpdated.clear();
}
});
requestAnimationFrame(f);
};
requestAnimationFrame(f);
}
subKey = '';
parent = null;
constructor(checker, data, _ = type(data, 'object')) {
// ...
// 생성할 때 한번은 notify 해주면
// subject 에 등록이 될 것이다.
ViewModel.notify(this);
Object.seal(this);
}
}

Observer#

뷰모델의 실질적인 주인은 바인더이다 바인더가 뷰모델을 구독한다.

바인더(옵저버)는 뷰모델 리스너이기도 하다 여러 역할을 하게 됐다. 인터페이스로 분리한다.

const Binder = class extends ViewModelListener {
#items = newSet;
#processors = {};
add(v, _ = type(v, BinderItem)) {
this.#items.add(v);
}
addProcessor(v, _0 = type(v, Processor)) {
this.#processors[v.cat] = v;
}
render(viewmodel, _ = type(viewmodel, ViewModel)) {
const processores = Object.entries(this.#processors);
this.#items.forEach(item => {
const vm = type(viewmodel[item.viewmodel], ViewModel), el = item.el;
processores.forEach(([pk, processor]) => {
Object.entries(vm[pk]).forEach(([k, v]) => {
processor.process(vm, el, k, v);
});
});
});
}
};

렌더할때도 process 를 했는데, 여기도 있다. 두곳이상의 알고리즘에 개입하는 외부에서 공급된 전략을 비지터라고 한다.

const Binder = class extends ViewModelListener {
#items = new Set;
#processors = {};
// 2) updated Set 에는 뷰모델 값이 들어있다. 서브키, 카테고리키, value,..
viewmodelUpdated(updated) {
const items = {};
// #items가 Set 이기 때문에 즉각적인 조회가 힘들어서??
// #items 을 items 로 환원시키려 한다. Set 은 forEach 밖에 못써서 obj 로 환
this.#items.forEach(item => {
// [래퍼객체, 아이템의 요소]
items[item.viewmodel] = [type(viewmodel[item.viewmodel], ViewModel), item.el];
});
updated.forEach(v => {
if (!items[v.subKey]) return;
const [vm, el] = items[v.subKey], processor = this.#processors[v.cat];
if (!el || !processor) return;
processor.process(vm, el, v.k, v.v);
});
}
add(v, _ = type(v, BinderItem)) {
this.#items.add(v);
}
addProcessor(v, _0 = type(v, Processor)) {
// ..
}
render(viewmodel, _ = type(viewmodel, ViewModel)) {
// ..
}
// 1) 바인더에서는 뷰모델을 받아서 자신의 리스너를 하고, 최초로 렌더를 해준다.
// 특정 뷰모델에 리스너로 등록되고 말고를 결정한다.
watch(viewmodel, _ = type(viewmodel, ViewModel)) {
viewmodel.addListener(this);
this.render(viewmodel);
}
unwatch(viewmodel, _ = type(viewmodel, ViewModel)) {
viewmodel.removeListener(this);
}
};
